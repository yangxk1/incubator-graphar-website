44 template <typename U, typename T>
45 static Status CastToLargeOffsetArray(
46  const std::shared_ptr<arrow::Array>& in,
47  const std::shared_ptr<arrow::DataType>& to_type,
48  std::shared_ptr<arrow::Array>& out) { // NOLINT(runtime/references)
49  auto array_data = in->data()->Copy();
50  auto offset = array_data->buffers[1];
51  using from_offset_type = typename U::offset_type;
52  using to_string_offset_type = typename T::offset_type;
53  auto raw_value_offsets_ =
54  offset == NULLPTR
55  ? NULLPTR
56  : reinterpret_cast<const from_offset_type*>(offset->data());
57  std::vector<to_string_offset_type> to_offset(offset->size() /
58  sizeof(from_offset_type));
59  for (size_t i = 0; i < to_offset.size(); ++i) {
60  to_offset[i] = raw_value_offsets_[i];
61  }
62  std::shared_ptr<arrow::Buffer> buffer;
63  arrow::TypedBufferBuilder<to_string_offset_type> buffer_builder;
64  RETURN_NOT_ARROW_OK(
65  buffer_builder.Append(to_offset.data(), to_offset.size()));
66  RETURN_NOT_ARROW_OK(buffer_builder.Finish(&buffer));
67  array_data->type = to_type;
68  array_data->buffers[1] = buffer;
69  out = arrow::MakeArray(array_data);
70  RETURN_NOT_ARROW_OK(out->ValidateFull());
71  return Status::OK();
72 }

74 template <typename U, typename T>
75 static Status CastToLargeOffsetArray(
76  const std::shared_ptr<arrow::ChunkedArray>& in,
77  const std::shared_ptr<arrow::DataType>& to_type,
78  std::shared_ptr<arrow::ChunkedArray>& out) { // NOLINT(runtime/references)
79  std::vector<std::shared_ptr<arrow::Array>> chunks;
80  for (auto const& chunk : in->chunks()) {
81  std::shared_ptr<arrow::Array> array;
82  auto status = CastToLargeOffsetArray<U, T>(chunk, to_type, array);
83  GAR_RETURN_NOT_OK(status);
84  chunks.emplace_back(array);
85  }
86  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(out, arrow::ChunkedArray::Make(chunks));
87  return Status::OK();
88 }

112 Result<std::shared_ptr<arrow::Table>> FileSystem::ReadFileToTable(
113  const std::string& path, FileType file_type,
114  const std::vector<int>& column_indices) const noexcept {
115  parquet::arrow::FileReaderBuilder builder;
116  auto open_file_status = builder.OpenFile(path);
117  if (!open_file_status.ok()) {
118  return Status::Invalid("Failed to open file: ", path, " - ",
119  open_file_status.ToString());
120  }
121  builder.memory_pool(arrow::default_memory_pool());
122  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(auto reader, builder.Build());
123  std::shared_ptr<arrow::Table> table;
124  if (column_indices.empty()) {
125  arrow::Status read_status = reader->ReadTable(&table);
126  if (!read_status.ok()) {
127  return Status::Invalid("Failed to read table from file: ", path, " - ",
128  read_status.ToString());
129  }
130  } else {
131  arrow::Status read_status = reader->ReadTable(column_indices, &table);
132  if (!read_status.ok()) {
133  return Status::Invalid("Failed to read table from file: ", path, " - ",
134  read_status.ToString());
135  }
136  }
137  return table;
138 }

140 Result<std::shared_ptr<arrow::Table>> FileSystem::ReadFileToTable(
141  const std::string& path, FileType file_type,
142  const util::FilterOptions& options) const noexcept {
143  std::shared_ptr<ds::FileFormat> format = GetFileFormat(file_type);
144  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(
145  auto factory, arrow::dataset::FileSystemDatasetFactory::Make(
146  arrow_fs_, {path}, format,
147  arrow::dataset::FileSystemFactoryOptions()));
148  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(auto dataset, factory->Finish());
149  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(auto scan_builder, dataset->NewScan());
150 
151  // Apply the row filter and select the specified columns
152  if (options.filter) {
153  GAR_ASSIGN_OR_RAISE(auto filter, options.filter->Evaluate());
154  RETURN_NOT_ARROW_OK(scan_builder->Filter(filter));
155  }
156  if (options.columns) {
157  RETURN_NOT_ARROW_OK(scan_builder->Project(*options.columns));
158  }
159 
160  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(auto scanner, scan_builder->Finish());
161  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(auto table, scanner->ToTable());
162  // cast string array to large string array as we need concatenate chunks in
163  // some places, e.g., in vineyard
164  for (int i = 0; i < table->num_columns(); ++i) {
165  std::shared_ptr<arrow::DataType> type = table->column(i)->type();
166  if (type->id() == arrow::Type::STRING) {
167  type = arrow::large_utf8();
168  } else if (type->id() == arrow::Type::BINARY) {
169  type = arrow::large_binary();
170  }
171  if (type->Equals(table->column(i)->type())) {
172  continue;
173  }
174  // do casting
175  auto field = table->field(i)->WithType(type);
176  std::shared_ptr<arrow::ChunkedArray> chunked_array;
177 
178  if (table->num_rows() == 0) {
179  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(
180  chunked_array, arrow::ChunkedArray::MakeEmpty(type));
181  } else if (type->Equals(arrow::large_utf8())) {
182  auto status = detail::CastToLargeOffsetArray<arrow::StringArray,
183  arrow::LargeStringArray>(
184  table->column(i), type, chunked_array);
185  GAR_RETURN_NOT_OK(status);
186  } else if (type->Equals(arrow::large_binary())) {
187  auto status = detail::CastToLargeOffsetArray<arrow::BinaryArray,
188  arrow::LargeBinaryArray>(
189  table->column(i), type, chunked_array);
190  GAR_RETURN_NOT_OK(status);
191  } else {
192  // noop
193  chunked_array = table->column(i);
194  }
195  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(table, table->RemoveColumn(i));
196  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(
197  table, table->AddColumn(i, field, chunked_array));
198  }
199  return table;
200 }

248 Status FileSystem::WriteTableToFile(
249  const std::shared_ptr<arrow::Table>& table, FileType file_type,
250  const std::string& path,
251  const std::shared_ptr<WriterOptions>& options) const noexcept {
252  // try to create the directory, oss filesystem may not support this, ignore
253  ARROW_UNUSED(arrow_fs_->CreateDir(path.substr(0, path.find_last_of("/"))));
254  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(auto output_stream,
255  arrow_fs_->OpenOutputStream(path));
256  switch (file_type) {
257  case FileType::CSV: {
258  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(
259  auto writer,
260  arrow::csv::MakeCSVWriter(output_stream.get(), table->schema(),
261  options->getCsvOption()));
262  RETURN_NOT_ARROW_OK(writer->WriteTable(*table));
263  RETURN_NOT_ARROW_OK(writer->Close());
264  break;
265  }
266  case FileType::PARQUET: {
267  auto schema = table->schema();
268  RETURN_NOT_ARROW_OK(parquet::arrow::WriteTable(
269  *table, arrow::default_memory_pool(), output_stream, 64 * 1024 * 1024,
270  options->getParquetWriterProperties(),
271  options->getArrowWriterProperties()));
272  break;
273  }
274 #ifdef ARROW_ORC
275  case FileType::ORC: {
276  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(
277  auto writer, arrow::adapters::orc::ORCFileWriter::Open(
278  output_stream.get(), options->getOrcOption()));
279  RETURN_NOT_ARROW_OK(writer->Write(*table));
280  RETURN_NOT_ARROW_OK(writer->Close());
281  break;
282  }
283 #endif
284  default:
285  return Status::Invalid(
286  "Unsupported file type: ", FileTypeToString(file_type), " for wrting.");
287  }
288  return Status::OK();
289 }

332 Result<std::shared_ptr<FileSystem>> FileSystemFromUriOrPath(
333  const std::string& uri_string, std::string* out_path) {
334  if (uri_string.length() >= 1 && uri_string[0] == '/') {
335  // if the uri_string is an absolute path, we need to create a local file
336  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(
337  auto arrow_fs,
338  arrow::fs::FileSystemFromUriOrPath(uri_string, out_path));
339  // arrow would delete the last slash, so use uri string
340  if (out_path != nullptr) {
341  *out_path = uri_string;
342  }
343  return std::make_shared<FileSystem>(arrow_fs);
344  }
345 
346  GAR_RETURN_ON_ARROW_ERROR_AND_ASSIGN(
347  auto arrow_fs, arrow::fs::FileSystemFromUriOrPath(uri_string));
348  auto uri = uri::parse_uri(uri_string);
349  if (uri.error != uri::Error::None) {
350  return Status::Invalid("Failed to parse URI: ", uri_string);
351  }
352  if (out_path != nullptr) {
353  if (uri.scheme == "file" || uri.scheme == "hdfs" || uri.scheme.empty()) {
354  *out_path = uri.path;
355  } else if (uri.scheme == "s3" || uri.scheme == "gs") {
356  // bucket name is the host, path is the path
357  *out_path = uri.authority.host + uri.path;
358  } else {
359  return Status::Invalid("Unrecognized filesystem type in URI: ",
360  uri_string);
361  }
362  }
363  return std::make_shared<FileSystem>(arrow_fs);
364 }

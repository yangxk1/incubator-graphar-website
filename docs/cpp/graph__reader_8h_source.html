171 class VertexIter {
172  public:
180  explicit VertexIter(const std::shared_ptr<VertexInfo>& vertex_info,
181  const std::string& prefix, IdType offset,
182  const std::vector<std::string>& labels,
183  const bool& is_filtered = false,
184  const std::vector<IdType>& filtered_ids = {}) noexcept {
185  if (!labels.empty()) {
186  labels_ = labels;
187  label_reader_ =
188  VertexPropertyArrowChunkReader(vertex_info, labels, prefix);
189  }
190  for (const auto& pg : vertex_info->GetPropertyGroups()) {
191  readers_.emplace_back(vertex_info, pg, prefix);
192  }
193  is_filtered_ = is_filtered;
194  filtered_ids_ = filtered_ids;
195  cur_offset_ = offset;
196  }
197

200  : readers_(other.readers_),
201  cur_offset_(other.cur_offset_),
202  labels_(other.labels_),
203  label_reader_(other.label_reader_),
204  is_filtered_(other.is_filtered_),
205  filtered_ids_(other.filtered_ids_) {}
206 
208  Vertex operator*() noexcept {
209  if (is_filtered_) {
210  for (auto& reader : readers_) {
211  reader.seek(filtered_ids_[cur_offset_]);
212  }
213  } else {
214  for (auto& reader : readers_) {
215  reader.seek(cur_offset_);
216  }
217  }
218 
219  return Vertex(cur_offset_, readers_);
220  }
221 
223  IdType id() {
224  if (is_filtered_) {
225  return filtered_ids_[cur_offset_];
226  } else {
227  return cur_offset_;
228  }
229  }
230 
232  template <typename T>
233  Result<T> property(const std::string& property) noexcept {
234  std::shared_ptr<arrow::ChunkedArray> column(nullptr);
235  if (is_filtered_) {
236  for (auto& reader : readers_) {
237  reader.seek(filtered_ids_[cur_offset_]);
238  GAR_ASSIGN_OR_RAISE(auto chunk_table,
239  reader.GetChunk(graphar::GetChunkVersion::V1));
240  column = util::GetArrowColumnByName(chunk_table, property);
241  if (column != nullptr) {
242  break;
243  }
244  }
245  } else {
246  for (auto& reader : readers_) {
247  reader.seek(cur_offset_);
248  GAR_ASSIGN_OR_RAISE(auto chunk_table,
249  reader.GetChunk(graphar::GetChunkVersion::V1));
250  column = util::GetArrowColumnByName(chunk_table, property);
251  if (column != nullptr) {
252  break;
253  }
254  }
255  }
256 
257  if (column != nullptr) {
258  auto array = util::GetArrowArrayByChunkIndex(column, 0);
259  GAR_ASSIGN_OR_RAISE(auto data, util::GetArrowArrayData(array));
260  return util::ValueGetter<T>::Value(data, 0);
261  }
262  return Status::KeyError("Property with name ", property,
263  " does not exist in the vertex.");
264  }

490 class EdgeIter {
491  public:
505  explicit EdgeIter(const std::shared_ptr<EdgeInfo>& edge_info,
506  const std::string& prefix, AdjListType adj_list_type,
507  IdType global_chunk_index, IdType offset,
508  IdType chunk_begin, IdType chunk_end,
509  std::shared_ptr<util::IndexConverter> index_converter)
510  : adj_list_reader_(edge_info, adj_list_type, prefix),
511  global_chunk_index_(global_chunk_index),
512  cur_offset_(offset),
513  chunk_size_(edge_info->GetChunkSize()),
514  src_chunk_size_(edge_info->GetSrcChunkSize()),
515  dst_chunk_size_(edge_info->GetDstChunkSize()),
516  num_row_of_chunk_(0),
517  chunk_begin_(chunk_begin),
518  chunk_end_(chunk_end),
519  adj_list_type_(adj_list_type),
520  index_converter_(index_converter) {
521  vertex_chunk_index_ =
522  index_converter->GlobalChunkIndexToIndexPair(global_chunk_index).first;
523  adj_list_reader_.seek_chunk_index(vertex_chunk_index_);
524  const auto& property_groups = edge_info->GetPropertyGroups();
525  for (const auto& pg : property_groups) {
526  property_readers_.emplace_back(edge_info, pg, adj_list_type, prefix),
527  property_readers_.back().seek_chunk_index(vertex_chunk_index_);
528  }
529  if (adj_list_type == AdjListType::ordered_by_source ||
530  adj_list_type == AdjListType::ordered_by_dest) {
531  offset_reader_ = std::make_shared<AdjListOffsetArrowChunkReader>(
532  edge_info, adj_list_type, prefix);
533  }
534  }

537  EdgeIter(const EdgeIter& other)

547  num_row_of_chunk_(other.num_row_of_chunk_),
548  chunk_begin_(other.chunk_begin_),
549  chunk_end_(other.chunk_end_),
550  adj_list_type_(other.adj_list_type_),
551  index_converter_(other.index_converter_) {}

560  }

569  template <typename T>
570  Result<T> property(const std::string& property) noexcept {
571  std::shared_ptr<arrow::ChunkedArray> column(nullptr);
572  for (auto& reader : property_readers_) {
573  reader.seek(cur_offset_);
574  GAR_ASSIGN_OR_RAISE(auto chunk_table, reader.GetChunk());
575  column = util::GetArrowColumnByName(chunk_table, property);
576  if (column != nullptr) {
577  break;
578  }
579  }
580  if (column != nullptr) {
581  auto array = util::GetArrowArrayByChunkIndex(column, 0);
582  GAR_ASSIGN_OR_RAISE(auto data, util::GetArrowArrayData(array));
583  return util::ValueGetter<T>::Value(data, 0);
584  }
585  return Status::KeyError("Property with name ", property,
586  " does not exist in the edge.");
587  }
588 
590  EdgeIter& operator++() {
591  if (num_row_of_chunk_ == 0) {
592  adj_list_reader_.seek(cur_offset_);
593  GAR_ASSIGN_OR_RAISE_ERROR(num_row_of_chunk_,
594  adj_list_reader_.GetRowNumOfChunk());
595  }
596  auto st = adj_list_reader_.seek(++cur_offset_);
597  if (st.ok() && num_row_of_chunk_ != chunk_size_) {
598  // check the row offset is overflow
599  auto row_offset = cur_offset_ % chunk_size_;
600  if (row_offset >= num_row_of_chunk_) {
601  cur_offset_ = (cur_offset_ / chunk_size_ + 1) * chunk_size_;
602  adj_list_reader_.seek(cur_offset_);
603  st =
604  Status::KeyError("The row offset is overflow, move to next chunk.");
605  }
606  }
607  if (st.ok() && num_row_of_chunk_ == chunk_size_ &&
608  cur_offset_ % chunk_size_ == 0) {
609  GAR_ASSIGN_OR_RAISE_ERROR(num_row_of_chunk_,
610  adj_list_reader_.GetRowNumOfChunk());
611  ++global_chunk_index_;
612  // The reader also need to be updated at the boundaries of chunks of size
613  // chunk_size.
614  for (auto& reader : property_readers_) {
615  reader.next_chunk();
616  }
617  }
618  if (st.IsKeyError()) {
619  st = adj_list_reader_.next_chunk();
620  ++global_chunk_index_;
621  ++vertex_chunk_index_;
622  if (!st.IsIndexError()) {
623  GAR_ASSIGN_OR_RAISE_ERROR(num_row_of_chunk_,
624  adj_list_reader_.GetRowNumOfChunk());
625  for (auto& reader : property_readers_) {
626  reader.next_chunk();
627  }
628  }
629  cur_offset_ = 0;
630  adj_list_reader_.seek(cur_offset_);
631  }
632  return *this;
633  }

702  void to_begin() {
703  global_chunk_index_ = chunk_begin_;
704  cur_offset_ = 0;

712  bool is_end() const { return global_chunk_index_ >= chunk_end_; }

715  bool next_src() {
716  if (is_end())
717  return false;
718  IdType id = this->source();
719  IdType pre_vertex_chunk_index = vertex_chunk_index_;
720  if (adj_list_type_ == AdjListType::ordered_by_source) {
721  this->operator++();
722  if (is_end() || this->source() != id)
723  return false;
724  else
725  return true;
726  }
727  this->operator++();
728  while (!is_end()) {
729  if (this->source() == id) {
730  return true;
731  }
732  if (adj_list_type_ == AdjListType::unordered_by_source) {
733  if (vertex_chunk_index_ > pre_vertex_chunk_index)
734  return false;
735  }
736  this->operator++();
737  }
738  return false;
739  }
740 
745  bool next_dst() {
746  if (is_end())
747  return false;
748  IdType id = this->destination();
749  IdType pre_vertex_chunk_index = vertex_chunk_index_;
750  if (adj_list_type_ == AdjListType::ordered_by_dest) {
751  this->operator++();
752  if (is_end() || this->destination() != id)
753  return false;
754  else
755  return true;
756  }
757  this->operator++();
758  while (!is_end()) {
759  if (this->destination() == id) {
760  return true;
761  }
762  if (adj_list_type_ == AdjListType::unordered_by_dest) {
763  if (vertex_chunk_index_ > pre_vertex_chunk_index)
764  return false;
765  }
766  this->operator++();
767  }
768  return false;
769  }

797  adj_list_reader_.seek(cur_offset_);
798  for (auto& reader : property_readers_) {
799  reader.seek_chunk_index(vertex_chunk_index_);
800  }
801  GAR_ASSIGN_OR_RAISE_ERROR(num_row_of_chunk_,
802  adj_list_reader_.GetRowNumOfChunk());
803  }

808  std::vector<AdjListPropertyArrowChunkReader> property_readers_;
